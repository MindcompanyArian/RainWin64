color backgroundColor = color(0, 0, 30);
int rainDrops = 100;

dropPhys[] dp = new dropPhys[rainDrops];

void setup() {
  size(800, 600, P3D);
  
  for (int i = 0; i < dp.length; i++) {
    dp[i] = new dropPhys();
  }
}

void draw() { 
  background (backgroundColor);
  noStroke();
  rect(width * 0.01, height - 160, width * 0.98, 160);
  fill(0, 97, 18);
  for (int i = 0; i < dp.length; i++) {
    dp[i].phys();
    dp[i].render();
  }
}
